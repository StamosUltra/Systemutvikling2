package application;

import java.util.ArrayList;

/**
 * Created by joaki on 21.09.2017.
 */
public class Menu {
    private ArrayList<Item> items;

    public Menu() {
    }

    public ArrayList<Item> getItems() {
        return items;
    }

    public int addItems(ArrayList<Item> is) {
        if (is != null) {
            int itemsAdded = 0;
            for (Item i : is) {
                if (this.items.add(i)) itemsAdded++;
            }
            return itemsAdded;
        }
        return -1;
    }

    /**
     * Removes items from the menu based on the name
     * @param names names of items you want to remove
     * @return number of items removed, -1 if names == null
     */
    public int removeItems(String[] names) {
        if (names != null) {
            int itemsRemoved = 0;
            for (Item i : this.items) {
                for (String name : names) {
                    if (name != null && name.trim().length() > 0) {
                        name = name.trim();
                        if (name.equals(i.getName())) {
                            if(this.items.remove(i)) itemsRemoved++;
                        }
                    }
                    else break;
                }
            }
            return itemsRemoved;
        }
        return -1;
    }
}
