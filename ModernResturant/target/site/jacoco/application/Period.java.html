package application;

import java.util.Date;

/**
 * Created by joaki on 21.09.2017.
 */
public class Period {
    private Date from;
    private Date to;

    public Period(Date from, Date to) {
        if (from == null || to == null || to.compareTo(from) <= 0) throw new IllegalArgumentException("to-date must come after from date");

        this.from = from;
        this.to = to;
    }

    public Date getFrom() {
        return from;
    }

    public Date getTo() {
        return to;
    }

    public boolean editDates(Date from, Date to) {
        if (from != null && to != null && to.compareTo(from) > 0) {
            this.from = from;
            this.to = to;
        }
        return false;
    }

    public boolean isInPeriod(Period period) {
        if (period != null) {
            if (period.getFrom().compareTo(this.from) >= 0 && period.getFrom().compareTo(this.to) < 0) return true; // if from is the same time as this.from or in between this.from and this.to
            if (period.getFrom().compareTo(this.from) <= 0 && period.getFrom().compareTo(this.from) >= 0) return true; // If from is earlier than this.from and to is in after this.from
            return false;
        }
        return false;
    }
}
